from typing import Iterable


def except_zero(items: list) -> Iterable:
    res = [i for i in enumerate(items) if i[1] == 0]
    res_2 = sorted(filter(lambda x: x > 0, items))
    for i in res:
        res_2.insert(i[0], i[1])
    return res_2
