from collections.abc import Iterable


def except_zero(items: list[int]) -> Iterable[int]:

    res = [i for i, v in enumerate(items) if v == 0]

from collections.abc import Iterable
import numpy


def except_zero(items: list[int]) -> Iterable[int]:

    items = numpy.r_[items]
    items[items!= 0] = sorted(items[items!= 0])

    return items
